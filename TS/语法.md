# TS 语法

## 类型注解

```ts
// 基本类型
// : number 表示类型注解
// 作用：添加类型约束
let age: number = 18;
```

```ts
// 数组类型
let nums1: number[] = [1, 3, 5]; // 推荐写法
let nums2: Array<number> = [1, 3, 5];

// 联合类型
let arr: (number | string)[] = [1, 'a', 3，‘b’]; // 支持多种类型
```

```ts
// 函数类型

// 单独指定参数类型，返回值类型
function add(num1: number, num2: number): number {
  return num1 + num2;
}

// 函数变量方式声明
const add = (num1: number, num2: number): number => {
  return num1 + num2;
};

// 同时指定参数类型，返回值类型
const add: (num1: number, num2: number) => number = (num1, num2) => {
  return num1 + num2;
};

// 函数可选参数,必选参数必须放在可选参数前面
const add = (num1?: number, num2?: number): void => {
  console.log('num1: ', num1, 'num2: ', num2);
};
```

```ts
// 对象类型
let person: {
  name: string;
  age: number;
  sayHi: () => void;
  greet: (name: string) => void;
} = {
  name: 'admin',
  age: 18,
  sayHi() {},
  greet(name) {},
};

// 可选参数
function axios(config: { url: string; method?: string }) {
  console.log('config', config);
}
```

```ts
// 接口类型

interface Person {
  name: string;
  age: number;
  sayHi: () => void;
}

t;
type Person = {
  name: string;
  age: number;
  sayHi: () => void;
};

let person: Person = {
  name: 'Jack',
  age: 18,
  sayHi() {},
};
```

```ts
// type关键字
type arrType = (number | string)[];
```
